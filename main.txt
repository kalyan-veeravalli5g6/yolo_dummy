import 'package:flutter/material.dart';
import 'package:ultralytics_yolo/ultralytics_yolo.dart';
import 'package:ultralytics_yolo/widgets/yolo_controller.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.deepPurple),
      ),
      home: const HomePage(),
    );
  }
}

class HomePage extends StatefulWidget {
  const HomePage({super.key});

  @override
  State<HomePage> createState() => _HomePageState();
}

class _HomePageState extends State<HomePage> {
  final _controller = YOLOViewController();
  bool isPaused = false;
  bool _isCameraActive = true;
  List<String> detectedItems = [];

  void _togglePause() {
    setState(() {
      isPaused = !isPaused;
      _isCameraActive = !_isCameraActive;
    });

    debugPrint(
      isPaused
          ? '⏸️ Detection Paused - Camera stopped'
          : '▶️ Detection Resumed - Camera started',
    );
  }

  void _onDetectionResult(List<dynamic>? results) {
    if (results == null || !_isCameraActive || isPaused) {
      return;
    }

    print("========================================");
    print("Detection Results:");

    setState(() {
      detectedItems = results
          .map((result) => "Class ${result.classIndex}: ${result.className}")
          .toList();
    });

    results.forEach((result) => print("  Class ${result.classIndex}"));
    print("========================================");
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('YOLO Detection'),
        actions: [
          IconButton(
            icon: Icon(isPaused ? Icons.play_arrow : Icons.pause),
            onPressed: _togglePause,
            tooltip: isPaused ? 'Resume' : 'Pause',
          ),
        ],
      ),
      body: Column(
        children: [
          _buildYOLOView(),
          const SizedBox(height: 16),
          Expanded(child: _buildDetectedItemsList()),
        ],
      ),
    );
  }

  Widget _buildYOLOView() {
    return GestureDetector(
      onTap: _togglePause,
      child: Container(
        margin: const EdgeInsets.all(8),
        height: MediaQuery.of(context).size.height * 0.48,
        decoration: BoxDecoration(
          borderRadius: BorderRadius.circular(12),
          border: Border.all(
            color: isPaused ? Colors.grey : Colors.deepPurple,
            width: isPaused ? 4 : 2,
          ),
        ),
        child: Stack(
          children: [
            if (isPaused || !_isCameraActive)
              _buildPausedOverlay()
            else
              ClipRRect(
                borderRadius: BorderRadius.circular(10),
                child: YOLOView(
                  key: ValueKey('yolo_$_isCameraActive'),
                  useGpu: false,
                  showOverlays: true,
                  controller: _controller,
                  task: YOLOTask.detect,
                  modelPath: "new",
                  showNativeUI: false,
                  streamingConfig: YOLOStreamingConfig.custom(
                    includeDetections: true,
                    throttleInterval: const Duration(milliseconds: 100),
                    maxFPS: 20,
                    inferenceFrequency: 10,
                  ),
                  onResult: _onDetectionResult,
                ),
              ),
            // Status Indicator
            Positioned(top: 15, right: 15, child: _buildStatusIndicator()),
          ],
        ),
      ),
    );
  }

  Widget _buildPausedOverlay() {
    return Container(
      decoration: BoxDecoration(
        color: Colors.black87,
        borderRadius: BorderRadius.circular(10),
      ),
      child: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            const Icon(
              Icons.pause_circle_outline,
              color: Colors.white,
              size: 80,
            ),
            const SizedBox(height: 16),
            const Text(
              'Detection Paused',
              style: TextStyle(
                color: Colors.white,
                fontSize: 24,
                fontWeight: FontWeight.bold,
              ),
            ),
            const SizedBox(height: 8),
            const Text(
              'Tap to Resume',
              style: TextStyle(color: Colors.white70, fontSize: 16),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildStatusIndicator() {
    return Container(
      padding: const EdgeInsets.symmetric(horizontal: 10, vertical: 6),
      decoration: BoxDecoration(
        color: isPaused ? Colors.red : Colors.green,
        borderRadius: BorderRadius.circular(20),
      ),
      child: Row(
        mainAxisSize: MainAxisSize.min,
        children: [
          Icon(
            isPaused ? Icons.pause : Icons.fiber_manual_record,
            color: Colors.white,
            size: 12,
          ),
          const SizedBox(width: 8),
          Text(
            isPaused ? 'PAUSED' : 'DETECTING',
            style: const TextStyle(
              color: Colors.white,
              fontSize: 12,
              fontWeight: FontWeight.bold,
            ),
          ),
        ],
      ),
    );
  }

  Widget _buildDetectedItemsList() {
    return Column(
      children: [
        const Padding(
          padding: EdgeInsets.symmetric(horizontal: 16),
          child: Align(
            alignment: Alignment.centerLeft,
            child: Text(
              'Detected Items',
              style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
            ),
          ),
        ),
        const SizedBox(height: 8),
        Expanded(
          child: detectedItems.isEmpty
              ? Center(
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      Icon(
                        isPaused ? Icons.pause_circle_outline : Icons.search,
                        size: 48,
                        color: Colors.grey,
                      ),
                      const SizedBox(height: 16),
                      Text(
                        isPaused
                            ? 'Detection is paused'
                            : 'No objects detected yet',
                        style: const TextStyle(
                          color: Colors.grey,
                          fontSize: 16,
                        ),
                      ),
                    ],
                  ),
                )
              : ListView.builder(
                  padding: const EdgeInsets.symmetric(horizontal: 16),
                  itemCount: detectedItems.length,
                  itemBuilder: (context, index) {
                    return Card(
                      margin: const EdgeInsets.only(bottom: 8),
                      child: ListTile(
                        leading: const Icon(
                          Icons.check_circle,
                          color: Colors.green,
                        ),
                        title: Text(detectedItems[index]),
                      ),
                    );
                  },
                ),
        ),
      ],
    );
  }

  @override
  void dispose() {
    super.dispose();
  }
}
